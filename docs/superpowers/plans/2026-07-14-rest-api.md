# REST API for PO and Contract Match Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Expose the existing PO/Contract validation pipeline as an HTTP API (`POST /validate`) so any external application can submit a PO PDF, a Contract PDF, and an SAP record over HTTP and get the discrepancy report back.

**Architecture:** A new `api/` package (`api/auth.py`, `api/main.py`) is a thin FastAPI layer that calls the existing, unchanged `run_extraction_crew`, `run_validation_crew`, and `write_report` functions. `main.py` (CLI) and `api/main.py` (server) are independent entry points into the same pipeline; neither imports the other.

**Tech Stack:** FastAPI, Uvicorn (ASGI server), python-multipart (file upload parsing), httpx (used internally by FastAPI's `TestClient`), Pydantic (already in use via `schemas/`).

## Global Constraints

- Reuse `run_extraction_crew`, `run_validation_crew`, `write_report` exactly as they are today — no changes to `crews/`, `agents/`, `utils/report_writer.py`, or `schemas/`.
- Exactly one endpoint: `POST /validate`. No `/extract`, `/status`, or job-queue endpoints.
- Synchronous request/response only — no background workers, no polling.
- Caller supplies `sap_record` per-request as a JSON string; the API never reads `config/sap_po_record.json` itself.
- Auth: header `X-API-Key`, checked against `API_KEY` read from the environment (`.env`). Missing or wrong key → `401`.
- Uploaded PDFs are saved to a temp directory and deleted after each request; `output/` JSON+PDF reports persist exactly as `main.py` produces them today.
- Response body: `{"sap_validation": ValidationReport, "contract_validation": ValidationReport, "report_pdf_base64": str}`.

```python
import base64
import json
import shutil
import tempfile
from pathlib import Path

from fastapi import Depends, FastAPI, File, Form, UploadFile
from pydantic import BaseModel

from api.auth import require_api_key
from crews.extraction_crew import run_extraction_crew
from crews.validation_crew import run_validation_crew
from schemas.discrepancy_schema import ValidationReport
from utils.report_writer import write_report

app = FastAPI(title="PO and Contract Match API")


class ValidateResponse(BaseModel):
    sap_validation: ValidationReport
    contract_validation: ValidationReport
    report_pdf_base64: str


@app.post(
    "/validate",
    response_model=ValidateResponse,
    dependencies=[Depends(require_api_key)],
)
async def validate(
    po_pdf: UploadFile = File(...),
    contract_pdf: UploadFile = File(...),
    sap_record: str = Form(...),
) -> ValidateResponse:
    sap_record_dict = json.loads(sap_record)

    temp_dir = Path(tempfile.mkdtemp(prefix="po_contract_match_"))
    try:
        po_path = temp_dir / "po.pdf"
        contract_path = temp_dir / "contract.pdf"
        po_path.write_bytes(await po_pdf.read())
        contract_path.write_bytes(await contract_pdf.read())

        po_details, contract_details = run_extraction_crew(str(po_path), str(contract_path))
        sap_report, contract_report = run_validation_crew(
            po_details, contract_details, sap_record_dict
        )
        report_json_path = write_report(sap_report, contract_report)

        pdf_path = report_json_path.with_suffix(".pdf")
        pdf_base64 = base64.b64encode(pdf_path.read_bytes()).decode("ascii")

        return ValidateResponse(
            sap_validation=sap_report,
            contract_validation=contract_report,
            report_pdf_base64=pdf_base64,
        )
    finally:
        shutil.rmtree(temp_dir, ignore_errors=True)
```

Full updated content of `api/main.py`:
```python
import base64
import json
import logging
import shutil
import tempfile
from pathlib import Path

from fastapi import Depends, FastAPI, File, Form, HTTPException, Request, UploadFile
from fastapi.exceptions import RequestValidationError
from fastapi.responses import JSONResponse
from pydantic import BaseModel

from api.auth import require_api_key
from crews.extraction_crew import run_extraction_crew
from crews.validation_crew import run_validation_crew
from schemas.discrepancy_schema import ValidationReport
from utils.report_writer import write_report

logger = logging.getLogger(__name__)

app = FastAPI(title="PO and Contract Match API")


class ValidateResponse(BaseModel):
    sap_validation: ValidationReport
    contract_validation: ValidationReport
    report_pdf_base64: str


@app.exception_handler(RequestValidationError)
async def validation_exception_handler(
    request: Request, exc: RequestValidationError
) -> JSONResponse:
    return JSONResponse(status_code=400, content={"detail": exc.errors()})


@app.post(
    "/validate",
    response_model=ValidateResponse,
    dependencies=[Depends(require_api_key)],
)
async def validate(
    po_pdf: UploadFile = File(...),
    contract_pdf: UploadFile = File(...),
    sap_record: str = Form(...),
) -> ValidateResponse:
    try:
        sap_record_dict = json.loads(sap_record)
    except json.JSONDecodeError:
        raise HTTPException(status_code=400, detail="sap_record must be valid JSON")

    temp_dir = Path(tempfile.mkdtemp(prefix="po_contract_match_"))
    try:
        po_path = temp_dir / "po.pdf"
        contract_path = temp_dir / "contract.pdf"
        po_path.write_bytes(await po_pdf.read())
        contract_path.write_bytes(await contract_pdf.read())

        try:
            po_details, contract_details = run_extraction_crew(str(po_path), str(contract_path))
            sap_report, contract_report = run_validation_crew(
                po_details, contract_details, sap_record_dict
            )
            report_json_path = write_report(sap_report, contract_report)
        except Exception:
            logger.exception("Pipeline failed while processing /validate request")
            raise HTTPException(status_code=500, detail="Validation pipeline failed")

        pdf_path = report_json_path.with_suffix(".pdf")
        pdf_base64 = base64.b64encode(pdf_path.read_bytes()).decode("ascii")

        return ValidateResponse(
            sap_validation=sap_report,
            contract_validation=contract_report,
            report_pdf_base64=pdf_base64,
        )
    finally:
        shutil.rmtree(temp_dir, ignore_errors=True)
```
